234  void set_derived_values(const Molecule& molecule, const AtomicBasisSet& aobasis) {
235 
236  for (size_t iatom = 0; iatom < molecule.natom(); iatom++) {
237  if (molecule.get_pseudo_atom(iatom)){
238  set_derived_value("pure_ae",false);
239  continue;
240  }
241  }
242  set_derived_value("aobasis",molecule.guess_file());;
243  const int n_core = molecule.n_core_orb_all();
244 
245 
246  std::vector<double> proto=get<std::vector<double> >("protocol");
247  proto.back()=get<double>("econv");
248  set_derived_value("protocol",proto);
249  set_derived_value("dconv",sqrt(get<double>("econv"))*0.1);
250 
251  double z = molecule.total_nuclear_charge();
252  const double charge=get<double>("charge");
253  int nelec = int(z - charge - n_core*2);
254  if (fabs(nelec+charge+n_core*2-z) > 1e-6) {
255  error("non-integer number of electrons?", nelec+charge+n_core*2-z);
256  }
257 
258  set_derived_value("nalpha",(nelec + nopen())/2);
259  set_derived_value("nbeta",(nelec - nopen())/2);
260 
261  if (nalpha() < 0) error("negative number of alpha electrons?", nalpha());
262  if (nbeta() < 0) error("negative number of beta electrons?", nbeta());
263  if ((nalpha()+nbeta()) != nelec) error("nalpha+nbeta != nelec", nalpha()+nbeta());
264  if (nalpha() != nbeta()) set_derived_value("spin_restricted",false);
265 
266  set_derived_value("nmo_alpha",nalpha() + nvalpha());
267  set_derived_value("nmo_beta",nbeta() + nvbeta());
268 
269  // Ensure we have enough basis functions to guess the requested
270  // number of states ... a minimal basis for a closed-shell atom
271  // might not have any functions for virtuals.
272  int nbf = aobasis.nbf(molecule);
273  if ((nmo_alpha()>nbf) or (nmo_beta()>nbf)) error("too few basis functions?", nbf);
274 // nmo_alpha = std::min(nbf,nmo_alpha);
275 // nmo_beta = std::min(nbf,nmo_beta);
276 // if (nalpha>nbf || nbeta>nbf) error("too few basis functions?", nbf);
277 // nvalpha = nmo_alpha - nalpha;
278 // nvbeta = nmo_beta - nbeta;
279 
280  // Unless overridden by the user use a cell big enough to
281  // have exp(-sqrt(2*I)*r) decay to 1e-6 with I=1ev=0.037Eh
282  // --> need 50 a.u. either side of the molecule
283  set_derived_value("l",molecule.bounding_cube() + 50.0);
284 
285  set_derived_value("lo",molecule.smallest_length_scale());
286 
287  // set highest possible point group for symmetry
288  if (do_localize()) set_derived_value("pointgroup",std::string("c1"));
289  else set_derived_value("pointgroup",molecule.pointgroup_);
290 
291  // above two lines will not override user input, so check input is sane
292  if (do_localize() and do_symmetry()) {
293  error("\n\nsymmetry and localization cannot be used at the same time\n"
294  "switch from local to canonical orbitals (keyword canon)\n\n");
295  }
296  }

494  // initializes CalculationParameters using the contents of stream \c f
495  void read(std::istream& f) {
496  position_stream(f, "dft");
497  std::string s;
498  xc_data = "lda";
499  protocol_data = madness::vector_factory(1e-4, 1e-6);
500 
501  while (f >> s) {
502  if (s == "end") {
503  break;
504  }
505  else if (s == "loadbal") {
506  f >> vnucextra >> loadbalparts;
507  }
508  else if (s == "charge") {
509  f >> charge;
510  }
511  else if (s == "smear") {

792  void set_molecular_info(const Molecule& molecule, const AtomicBasisSet& aobasis, unsigned int n_core) {
793  double z = molecule.total_nuclear_charge();
794  int nelec = int(z - charge - n_core*2);
795  if (fabs(nelec+charge+n_core*2-z) > 1e-6) {
796  error("non-integer number of electrons?", nelec+charge+n_core*2-z);
797  }
798  nalpha = (nelec + nopen)/2;
799  nbeta = (nelec - nopen)/2;
800  if (nalpha < 0) error("negative number of alpha electrons?", nalpha);
801  if (nbeta < 0) error("negative number of beta electrons?", nbeta);
802  if ((nalpha+nbeta) != nelec) error("nalpha+nbeta != nelec", nalpha+nbeta);
803  nmo_alpha = nalpha + nvalpha;
804  nmo_beta = nbeta + nvbeta;
805  if (nalpha != nbeta) spin_restricted = false;
806 
807  // Ensure we have enough basis functions to guess the requested
808  // number of states ... a minimal basis for a closed-shell atom
809  // might not have any functions for virtuals.
810  int nbf = aobasis.nbf(molecule);
811  nmo_alpha = std::min(nbf,nmo_alpha);
812  nmo_beta = std::min(nbf,nmo_beta);
813  if (nalpha>nbf || nbeta>nbf) error("too few basis functions?", nbf);
814  nvalpha = nmo_alpha - nalpha;
815  nvbeta = nmo_beta - nbeta;
816 
817  // Unless overridden by the user use a cell big enough to
818  // have exp(-sqrt(2*I)*r) decay to 1e-6 with I=1ev=0.037Eh
819  // --> need 50 a.u. either side of the molecule
820 
821  if (L == 0.0) {
822  L = molecule.bounding_cube() + 50.0;
823  }
824 
825  lo = molecule.smallest_length_scale();
826 
827  // set molecular and computational point groups
828  // use highest point group unless specified by user
829 
830  // complain if symmetry has been set to anything other than c1
831  if ((symmetry!="default" and symmetry!="c1") and localize) {
832  error("\n\nsymmetry and localization cannot be used at the same time\n"
833  "switch from local to canonical orbitals (keyword canon)\n\n");
834  }
835 
836  // no symmetry keyword specified
837  if (symmetry=="default") {
838  if (localize) symmetry="c1";
839  else symmetry=molecule.pointgroup_;
840 
841  // symmetry keyword specified without pointgroup
842  } else if (symmetry=="full") {
843  symmetry=molecule.pointgroup_;
844  }

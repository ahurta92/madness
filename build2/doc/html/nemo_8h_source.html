98  /// normalize the nemos
99  template<typename T, std::size_t NDIM>
100  void normalize(std::vector<Function<T,NDIM> >& nemo,
101  const Function<double,NDIM> metric=Function<double,NDIM>()) const {
102 
103  // compute the norm of the reconstructed orbitals, includes the factor
104  std::vector<Function<T,NDIM> > mos = (metric.is_initialized()) ? metric*nemo : nemo;
105  std::vector<double> norms = norm2s(world, mos);
106 
107  // scale the nemos, excludes the nuclear correlation factor
108  std::vector<double> invnorm(norms.size());
109  for (std::size_t i = 0; i < norms.size(); ++i)
110  invnorm[i] = 1.0 / norms[i];
111  scale(world, nemo, invnorm);
112  }
113 
114  template<typename T>
115  static Tensor<T> Q2(const Tensor<T>& s) {
116  Tensor<T> Q = -0.5*s;
117  for (int i=0; i<s.dim(0); ++i) Q(i,i) += 1.5;
118  return Q;
119  }
120 
121  /// orthonormalize the vectors
122  template<typename T, std::size_t NDIM>
123  void orthonormalize(std::vector<Function<T,NDIM> >& nemo,
124  const Function<double,NDIM> metric=Function<double,NDIM>(),
125  const double trantol=FunctionDefaults<NDIM>::get_thresh()*0.01) const {
126 
127  if (nemo.size()==0) return;
128  normalize(nemo,metric);
129  double maxq;
130  do {
131  std::vector<Function<T,NDIM> > Rnemo = (metric.is_initialized()) ? metric*nemo : nemo;
132  Tensor<T> Q = Q2(matrix_inner(world, Rnemo, Rnemo));
133  maxq=0.0;
134  for (int i=0; i<Q.dim(0); ++i)
135  for (int j=0; j<i; ++j)
136  maxq = std::max(maxq,std::abs(Q(i,j)));
137 
138  Q.screen(trantol); // ???? Is this really needed?
139  nemo = transform(world, nemo, Q, trantol, true);
140  truncate(world, nemo);
141 // if (world.rank() == 0) print("ORTHOG2: maxq trantol", maxq, trantol);
142 
143  } while (maxq>0.01);
144  normalize(nemo,metric);
145  }

148  real_function_3d compute_density(const std::vector<Function<T,NDIM> > nemo) const {
149  return sum(world,abssq(world,nemo)).truncate();
150  }

350  /// the Laplacian of the density
351 
352  /// The Laplacian should currently only be used for subsequent convolution
353  /// with a Green's function (which is reasonably stable), but not on its own!
354  ///
355  /// The Laplacian of the cuspy density is numerically fairly unstable:
356  /// - a singular term may be rewritten using the nuclear potential (see below)
357  /// - the Laplacian of the regularized density is still very noisy
358  ///
359  /// It may be computed as
360  /// \f[
361  /// \Delta \rho = \Delta (R^2 \rho_R)
362  /// = \Delta (R^2) \rho_R + 2\nabla R \nabla \rho_R + R^2 \Delta \rho_R
363  /// = 2 R^2 U1^2 \rho_R -4 R^2 ( U-V ) \rho_R + R^2 \Delta\rho_R
364  /// \f]
365  /// where we can use the identity
366  /// \f[
367  /// U=V + R^{-1}[T,R]
368  /// -2 R (U-V) = \Delta R + 2\nabla R\dot \nabla
369  /// \f]
370  /// first term comes from the definition of the U potential as the commutator
371  /// over the kinetic energy (aka the Laplacian)
372  /// @param[in] rhonemo the regularized density \rho_R
373  /// @return the laplacian of the reconstructed density \Delta (R^2\rho_R)
374  real_function_3d make_laplacian_density(const real_function_3d& rhonemo) const;
375 
376  /// compute the kinetic energy potential using Eq. (16) of
377  /// R. A. King and N. C. Handy, “Kinetic energy functionals from the Kohn–Sham potential,”
378  /// Phys. Chem. Chem. Phys., vol. 2, no. 22, pp. 5049–5056, 2000.
379  real_function_3d kinetic_energy_potential(const vecfuncT& nemo) const;

382  /// smooth a function by projecting it onto k-1 and then average with k
383 
384  /// kept it here for further testing
385  static void smoothen(real_function_3d& f) {
386  int k=f.get_impl()->get_k();
387  real_function_3d fproj=project(f,k-1);
388  real_function_3d freproj=project(fproj,k);

604 /// rotate the KAIN subspace (cf. SCF.cc)
605 template<typename solverT>
606 void Nemo::rotate_subspace(World& world, const tensorT& U, solverT& solver,
607  int lo, int nfunc) const {
608  std::vector < std::vector<Function<double, 3> > > &ulist = solver.get_ulist();
609  std::vector < std::vector<Function<double, 3> > > &rlist = solver.get_rlist();
610  for (unsigned int iter = 0; iter < ulist.size(); ++iter) {
611  vecfuncT& v = ulist[iter];
612  vecfuncT& r = rlist[iter];

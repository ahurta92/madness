344  /// Default implementation of the pre/postamble for type checking.
345 
346  /// \tparam Archive The archive class.
347  /// \tparam T The type to serialized or to expect upon deserialization.
348  template <class Archive, class T>
349  struct ArchivePrePostImpl {
350  /// Deserialize a cookie and check the type.
351 
352  /// \param[in] ar The archive.
353  static inline void preamble_load(const Archive& ar) {
354  unsigned char ck = archive_typeinfo<T>::cookie;
355  unsigned char cookie;
356  ar.load(&cookie, 1); // cannot use >>
357  if (cookie != ck) {
358  char msg[255];
359  std::sprintf(msg,"InputArchive type mismatch: expected cookie "
360  "%u (%s) but got %u (%s) instead",
361  ck, archive_type_names[ck],
362  cookie,archive_type_names[cookie]);
363  std::cerr << msg << std::endl;
364  MADNESS_EXCEPTION(msg, static_cast<int>(cookie));
365  }
366  else {
367  MAD_ARCHIVE_DEBUG(std::cout << "read cookie " << archive_type_names[cookie] << std::endl);
368  }
369  }
370 
371  /// Serialize a cookie for type checking.
372 
373  /// \param[in] ar The archive.
374  static inline void preamble_store(const Archive& ar) {
375  unsigned char ck = archive_typeinfo<T>::cookie;
376  ar.store(&ck, 1); // cannot use <<
377  MAD_ARCHIVE_DEBUG(std::cout << "wrote cookie " << archive_type_names[ck] << std::endl);
378  }

79  /// This defines stuff that is serialiable by bitwise copy.
80  /// \warning This reports true for \c T that is an aggregate type
81  /// (struct or array) that includes pointers.
82  template <typename T>
83  struct is_trivially_serializable {
84  static const bool value = \
85  std::is_arithmetic<T>::value || \
86  std::is_member_function_pointer<T>::value || \
87  std::is_function<T>::value || \
88  std::is_function<typename std::remove_pointer<T>::type>::value || \
89  (std::is_pod<T>::value && !std::is_pointer<T>::value);
90 // ((std::is_class<T>::value || std::is_array<T>::value) && std::is_trivially_copyable<T>::value);

93  /// True for types that are "serialiable" to a std::ostream
94  template <typename T, typename = void>
95  struct is_ostreammable : std::false_type {};
96  template <typename T>
97  struct is_ostreammable<T, meta::void_t<decltype(std::declval<std::ostream&>() << std::declval<const T&>())>> : std::true_type {};
98  template <typename T> constexpr bool is_ostreammable_v = is_ostreammable<T>::value;
99  /// True for types that are "deserialiable" from an std::istream
100  template <typename T, typename = void>
101  struct is_istreammable : std::false_type {};
102  template <typename T>
103  struct is_istreammable<T, meta::void_t<decltype(std::declval<std::istream&>() >> std::declval<T&>())>> : std::true_type {};
104  template <typename T> constexpr bool is_istreammable_v = is_istreammable<T>::value;
105  /// providing automatic support for serializing to/from std streams requires bidirectional streammability
106  template <typename T> constexpr bool is_iostreammable_v = is_istreammable_v<T> && is_ostreammable_v<T>;

94 inline double mask1(double x) {
95  /* Iterated first beta function to switch smoothly
96  from 0->1 in [0,1]. n iterations produce 2*n-1
97  zero derivatives at the end points. Order of polyn
98  is 3^n.
99 
100  Currently use one iteration so that first deriv.
101  is zero at interior boundary and is exactly representable
102  by low order multiwavelet without refinement */
103 
104  x = (x*x*(3.-2.*x));
105  return x;
106 }

108 static double mask3(const coordT& ruser) {
109  coordT rsim;
110  user_to_sim(ruser, rsim);
111  double x= rsim[0], y=rsim[1], z=rsim[2];
112  double lo = 0.0625, hi = 1.0-lo, result = 1.0;
113  double rlo = 1.0/lo;
114 
115  if (x<lo)
116  result *= mask1(x*rlo);
117  else if (x>hi)
118  result *= mask1((1.0-x)*rlo);
119  if (y<lo)
120  result *= mask1(y*rlo);
121  else if (y>hi)
122  result *= mask1((1.0-y)*rlo);
123  if (z<lo)
124  result *= mask1(z*rlo);
125  else if (z>hi)
126  result *= mask1((1.0-z)*rlo);
127 
128  return result;
129 }

259 /// A MADNESS functor to compute the cartesian moment x^i * y^j * z^k (i, j, k integer and >= 0)
260 class MomentFunctor : public FunctionFunctorInterface<double,3> {
261 private:
262  const int i, j, k;
263 public:
264  MomentFunctor(int i, int j, int k) : i(i), j(j), k(k) {}
265  MomentFunctor(const std::vector<int>& x) : i(x[0]), j(x[1]), k(x[2]) {}
266  double operator()(const coordT& r) const {
267  double xi=1.0, yj=1.0, zk=1.0;
268  for (int p=0; p<i; ++p) xi *= r[0];
269  for (int p=0; p<j; ++p) yj *= r[1];
270  for (int p=0; p<k; ++p) zk *= r[2];
271  return xi*yj*zk;
272  }
273 };

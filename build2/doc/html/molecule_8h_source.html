54 class Atom {
55 public:
56  double x, y, z, q; ///< Coordinates and charge in atomic units
57  unsigned int atomic_number; ///< Atomic number
58  double mass; ///< Mass
59  bool pseudo_atom; ///< Indicates if this atom uses a pseudopotential
60 
61  explicit Atom(double x, double y, double z, double q, unsigned int atomic_number, bool pseudo_atom)
62  : x(x), y(y), z(z), q(q), atomic_number(atomic_number), pseudo_atom(pseudo_atom) {
63  mass= get_atomic_data(atomic_number).mass;
64 
65  if (mass==-1.0) MADNESS_EXCEPTION("faulty element in Atom",1);
66 
67  // unstable elements are indicated by negative masses, the mass
68  // is taken from the longest-living element
69  if (mass<0.0) mass*=-1.0;
70 
71  }
72 
73  explicit Atom(double x, double y, double z, double q, unsigned int atomic_number)
74  : x(x), y(y), z(z), q(q), atomic_number(atomic_number) {
75  mass= get_atomic_data(atomic_number).mass;
76 
77  if (mass==-1.0) MADNESS_EXCEPTION("faulty element in Atom",1);
78 
79  // unstable elements are indicated by negative masses, the mass
80  // is taken from the longest-living element
81  if (mass<0.0) mass*=-1.0;
82 
83  pseudo_atom = false;
84 
85  }
86 
87  Atom(const Atom& a) : x(a.x), y(a.y), z(a.z), q(a.q),
88  atomic_number(a.atomic_number), mass(a.mass), pseudo_atom(a.pseudo_atom) {}
89 
90  /// Default construct makes a zero charge ghost atom at origin
91  Atom() : x(0), y(0), z(0), q(0), atomic_number(0), mass(0.0), pseudo_atom(false) {}
92 
93  int get_atomic_number() const {return atomic_number;}
94 
95  madness::Vector<double,3> get_coords() const {
96  return madness::Vector<double,3>{x, y, z};
97  }
98 
99  /// return the mass in atomic units (electron mass = 1 a.u.)
100  double get_mass_in_au() const {return constants::atomic_mass_in_au * mass;}
101 
102  template <typename Archive>
103  void serialize(Archive& ar) {
104  ar & x & y & z & q & atomic_number & mass & pseudo_atom;
105  }
106 };

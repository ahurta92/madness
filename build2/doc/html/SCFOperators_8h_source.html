68  }

325  }
326 
327  template<typename T, std::size_t NDIM>
328  std::vector<complex_function_3d> operator()(const std::vector<Function<T,NDIM> >& vket) const {
329 
330  // the operator in cartesian components as
331  // L_z = - i (x del_y - y del_x)
332 
333  if (vket.size()==0) return std::vector<complex_function_3d>(0);
334 
335  real_function_3d x=real_factory_3d(world).functor([] (const coord_3d& r) {return r[0];});
336  real_function_3d y=real_factory_3d(world).functor([] (const coord_3d& r) {return r[1];});
337 
338  Derivative<T,NDIM> Dx = free_space_derivative<T,NDIM>(world, 0);
339  Derivative<T,NDIM> Dy = free_space_derivative<T,NDIM>(world, 1);
340  if (use_bsplines) {
341  Dx.set_bspline1();
342  Dy.set_bspline1();
343  }
344 
345  reconstruct(world,vket,true);
346  std::vector<Function<T,NDIM> > delx=apply(world,Dx,vket,false);
347  std::vector<Function<T,NDIM> > dely=apply(world,Dy,vket,true);
348 
349  std::vector<Function<T,NDIM> > result1=x*dely - y*delx;
350  std::vector<complex_function_3d> cresult1=convert<T,double_complex,NDIM>(world,result1);
351  std::vector<complex_function_3d> result=double_complex(0.0,-1.0)*cresult1;
352  return result;
353  }
354

356  double_complex operator()(const Function<T,NDIM>& bra, const Function<T,NDIM>& ket) const {
357  return inner(bra,this->operator()(ket));
358  }

667  real_function_3d result;
668  return result;
669  }
670  double operator()(const real_function_3d& bra, const real_function_3d& ket) const {
671  const auto compute_K = (scale_K != 0.0);
672  double J_00 = J(bra,ket);
673  double K_00 = compute_K ? (scale_K == 1. ? K(bra,ket) : scale_K * K(bra,ket)) : 0;
674  double T_00 = T(bra,ket);
675  double V_00 = V(bra,ket);

677  }

684  Tensor<double> tmat=T(vbra,vket);
685  Tensor<double> vmat=V(vbra,vket);

687  wtime+=wall_time(); ctime+=cpu_time();
688  if (world.rank()==0) printf("timer: %20.20s %8.2fs %8.2fs\n", "fock matrix", wtime, ctime);

18 /// index in packed lower triangle matrix, row >= col
19 template <typename Int>
20 size_t tridx(Int row, Int col) {
21  return (row < col) ? (row + col * (col + 1) / 2)
22  : (col + row * (row + 1) / 2);
23 }

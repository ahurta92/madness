115  friend bool similar(const MolecularOrbitals& mo1, const MolecularOrbitals& mo2, const double thresh=1.e-6) {
116 
117  if (mo1.mo.size()!=mo2.mo.size()) return false;
118  if (mo1.mo.size()==0) return true;
119 
120  World& world=mo1.mo.front().world();
121  bool similar=((mo1.eps-mo2.eps).normf()<thresh);
122  similar=similar and (norm2(world,mo1.mo-mo2.mo)<thresh);
123  similar=similar and (mo1.irreps==mo2.irreps);
124  similar=similar and (mo1.localize_sets==mo2.localize_sets);
125  return similar;
126  }

167

169

172  ar & eps & occ & localize_sets;
173  mo.resize(nmo);
174  for (unsigned int i = 0; i < mo.size(); ++i)
175  ar & mo[i];
176  unsigned int n_core = molecule.n_core_orb_all();
177  if (nmo > nmo_from_input) {
178  localize_sets = vector<int>(localize_sets.begin() + n_core,
179  localize_sets.begin() + n_core + nmo_from_input);
180  mo = std::vector<Function<T,NDIM> >(mo.begin() + n_core,
181  mo.begin() + n_core + nmo_from_input);
182  eps = copy(eps(Slice(n_core, n_core + nmo_from_input - 1)));
183  occ = copy(occ(Slice(n_core, n_core + nmo_from_input - 1)));
